Copy by SSH
resource "null_resource" "copy_files_to_server" {
    connection {
        type                = "ssh"
        user                = 
        password            = 
        host                = remote_server_ip
    }

    provisioner "file" {
      source      = local_file
      destination = remote_file
    }
}
