#3. list ethrnet devices 

  lspci | grep Ethernet

#4 tcpdump 
  #4.1 capture inbound traffic only
  
    sudo tcpdump -ni enp3s0 -Q in

    -n: IP instead of domain
    -i: interface
    -Q: direction: in, out, inout

  #4.2 capture traffic with lay2 information, such MAC or VLAN

    sudo tcpdump -ni enp3s0 host 8.8.8.8 -e

    -e: display layer2, including MAC or VLAN

  #5.1 display current number of active connections
