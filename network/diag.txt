#4 tcpdump 
  #4.1 capture inbound traffic only
  
    sudo tcpdump -ni enp3s0 -Q in

    -n: IP instead of domain
    -i: interface
    -Q: direction: in, out, inout

  #4.2 capture traffic with lay2 information, such MAC or VLAN

    sudo tcpdump -ni enp3s0 host 8.8.8.8 -e

    -e: display layer2, including MAC or VLAN

#5. display linux network connections, including TCP, UDP, ICMP, SCTP, or Other Layer 4+ protocols

  #5.1 display current number of active connections 

    cat /proc/sys/net/netfilter/nf_conntrack_count

  #5.2 display max number of connections
     cat /proc/sys/net/netfilter/nf_conntrack_max

  #5.3 display active TCP connections only. wants udp, replace tcp below to udp

    sudo conntrack -L | grep tcp | wc -l

  #5.4  connections usage percentage
    echo $(( $(cat /proc/sys/net/netfilter/nf_conntrack_count) * 100 / $(cat /proc/sys/net/netfilter/nf_conntrack_max) ))%

  #5.5 all tcp sessions

    sudo conntrack -L -p tcp | awk '{for(i=1;i<=NF;i++){if($i=="[ASSURED]"){state="ESTABLISHED"} else if($i~/^[A-Z_]+$/){state=$i}} if(state!="") counts[state]++} END{for(s in counts) print s, counts[s]}'
